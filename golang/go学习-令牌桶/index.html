Go学习 令牌桶

首先看个简单例子

package main

import (
	"fmt"
	"github.com/gin-gonic/gin"
	"golang.org/x/time/rate"
	"net/http"
	"time"
        "errors"
        "io/ioutil"
)

var (
	limit = rate.NewLimiter(rate.Limit(1), 2)
)

func limitMiddleware() func(*gin.Context) {
	return func(ctx *gin.Context) {
		if !limit.Allow() {
			ctx.String(http.StatusServiceUnavailable, "服务器忙...")
			ctx.Abort()
			return
		}
		ctx.Next()
	}
}
func GetOnce(addr string) ([]byte,error) {
	resp, err := http.Get("http://"+addr)
	if err != nil {
		return nil, err
	}
	defer resp.Body.Close()
        b,err := ioutil.ReadAll(resp.Body)
	if err != nil {
		return nil, err
	}
	if resp.StatusCode != http.StatusOK {
		return nil ,errors.New(string(b))
	}
	return b,nil
}

func clientGin(n int ,url string) {
	for i := 0; i < n; i++ {
		b, err := GetOnce(url)
		if err != nil {
			fmt.Printf("#%v: %v\n", i, err)
			time.Sleep(time.Millisecond * 200)
			continue
		}
		fmt.Printf("#%v: %v\n", i, string(b))
	}
}

func main() {
	r := gin.Default()
	r.Use(limitMiddleware())
	r.GET("/limit", func(c *gin.Context) {
		c.String(http.StatusOK, "ok")
	})
	go clientGin(100,"localhost:8080/limit")
	r.Run()
}

以上就是 go官方提供的令牌桶的实现, 简单好用

接下来我们详细看下源码实现

Limiter

我们从上面例子中的初始化函数入手

// limit 字段表示qps, burst 表示桶的大小
func NewLimiter(r Limit, b int) *Limiter {
	return &Limiter{
		limit: r,
		burst: b,
	}
}

Limiter struct

type Limiter struct {
	mu     sync.Mutex      // 锁
	limit  Limit           // qps
	burst  int             // 桶的容量
	tokens float64         // 剩余的tokens数
	last time.Time         // 最近发放token的时间
	lastEvent time.Time    // 最近限流的时间
}

Allow() 函数

// 对AllowN 包装, 获取一个token
func (lim *Limiter) Allow() bool {
	return lim.AllowN(time.Now(), 1)
}
// 对reserveN 包装, 延迟时间是0
func (lim *Limiter) AllowN(now time.Time, n int) bool {
	return lim.reserveN(now, n, 0).ok
}

func (lim *Limiter) reserveN(now time.Time, n int, maxFutureReserve time.Duration) Reservation {
        // 限流器的字段修改前要加锁的
	lim.mu.Lock()
	defer lim.mu.Unlock()
        // Inf 是 math.MaxFloat64
	if lim.limit == Inf {
		return Reservation{
                        // 这里ok字段直接返回true
			ok:        true,
			lim:       lim,
			tokens:    n,
			timeToAct: now,
		}
        // 如果这里 qps设置为0
	} else if lim.limit == 0 {
		var ok bool
                // 判断桶里面是否有存货
		if lim.burst >= n {
                        // 有存货直接设置为true
			ok = true
                        // 并且减少存货
			lim.burst -= n
		}
		return Reservation{
			ok:        ok,
			lim:       lim,
			tokens:    lim.burst,
			timeToAct: now,
		}
	}
        // 如果 0 < qps < math.MaxFloat64 ,
        // 那么就计算当前时间now距离last时间内 , 我们一共可以有多少个令牌发放, tokens最大只能是burst
        // 详细见 "说明1"
	now, last, tokens := lim.advance(now)

	// 减去本次要发的令牌数
	tokens -= float64(n)

	// 计算要等待的时间
	var waitDuration time.Duration
        // 如果没有足够的token可以发放
	if tokens < 0 {
                // 计算大概要多久才能有足够的token发放
                // (-tokens)/qps 秒
		waitDuration = lim.limit.durationFromTokens(-tokens)
	}

	// 桶的容量 要大于n 并且 等待的时间 小于等于 maxFutureReserve(这里是0)
        // 在这里的情况, 如果要ok=true的话 就必须burst大于等于n, 而且tokens也要大于等于n,这样waitDuration才大于等于0呢
        // 当然其他函数调用 reserveN 时 maxFutureReserve 可能不通
	ok := n <= lim.burst && waitDuration <= maxFutureReserve

	// Prepare reservation
	r := Reservation{
		ok:    ok,
		lim:   lim,
		limit: lim.limit,
	}
	if ok {
		r.tokens = n
                // 在maxFutureReserve 非0的时候, waitDuration 也可能非0, 所以可能要延迟
		r.timeToAct = now.Add(waitDuration)
	}

	// 更新状态
	if ok {
		lim.last = now
		lim.tokens = tokens
		lim.lastEvent = r.timeToAct
        // 没拿到token, 还是还原数据
	} else {
		lim.last = last
	}

	return r
}

看完我们发现, 令牌桶的设计就是一个 tokens 来计数, 通过 now时间last的时间差 , qps, burst 来计算和调整新的tokens 数

接着我们看一下另外一个函数, Wait()

Wait() 函数

Wait 顾名思义是会阻塞一直等待获得到token, 而Allow() 则直接返回 bool

Wait 还可以传入一个 ctx参数, 可以设置超时时间 或者结束时间

// 包装WaitN
func (lim *Limiter) Wait(ctx context.Context) (err error) {
        // 同样获取1个token
	return lim.WaitN(ctx, 1)
}

func (lim *Limiter) WaitN(ctx context.Context, n int) (err error) {
        // 同样需要锁
	lim.mu.Lock()
        // 读取桶大小和qps
	burst := lim.burst
	limit := lim.limit
	lim.mu.Unlock()

	if n > burst && limit != Inf {
		return fmt.Errorf("rate: Wait(n=%d) exceeds limiter's burst %d", n, burst)
	}
	// 查看ctx是否已经结束
	select {
	case <-ctx.Done():
		return ctx.Err()
	default:
	}
	// Determine wait limit
	now := time.Now()
        // InfDuration = 1<<63-1
	waitLimit := InfDuration
        // deadline 就是 ctx设置的
	if deadline, ok := ctx.Deadline(); ok {
                // waitLimit 表示ctx 结束时间
		waitLimit = deadline.Sub(now)
	}
	// 这里调用reserveN 和Allow()不同的是, 延迟时间参数waitLimit
        // waitLimit 既然表示ctx cancel() 的最后时间, 那么意思是 waitDuration 只要小于等于 结束时间 就有可能 r.ok == true
	r := lim.reserveN(now, n, waitLimit)
	if !r.ok {
		return fmt.Errorf("rate: Wait(n=%d) would exceed context deadline", n)
	}
	// 这里是把 r.timeToAct - now, r.timeToAct 我们知道在reserveN 函数内部被更新了
	delay := r.DelayFrom(now)
	if delay == 0 {
		return nil
	}
        // 新建一个定时器
	t := time.NewTimer(delay)
	defer t.Stop()
	select {
        // 延迟
	case <-t.C:
		// We can proceed.
		return nil
        // 也有可能ctx 被设置结束了
	case <-ctx.Done():
                // Reservation 是包含了 limiter的
                // 通过Reservation对令牌桶的数据进行还原, 详细见 "说明2"
		r.Cancel()
		return ctx.Err()
	}
}

详细说明

说明1

func (lim *Limiter) advance(now time.Time) (newNow time.Time, newLast time.Time, newTokens float64) {
        // 最近发放token的时间
	last := lim.last
        // 如果 now < last
	if now.Before(last) {
		last = now
	}
        // 到这里 now 肯定是 >= last
	// now - last >= 0
	elapsed := now.Sub(last)
        // 获得这段时间中 可以有几个令牌
	delta := lim.limit.tokensFromDuration(elapsed)
        // 获得最新可以发放的总令牌数
	tokens := lim.tokens + delta
        // 如果超过了桶的容量, 那没办法, 最大是桶的容量
	if burst := float64(lim.burst); tokens > burst {
		tokens = burst
	}
	return now, last, tokens
}

// 比如qps=5, 表示1秒可以有5个token
// 那假如 now-last 差是1s
// 那么说明1s我没有发放token了, 那我应该要发放 5个token
func (limit Limit) tokensFromDuration(d time.Duration) float64 {
	if limit <= 0 {
		return 0
	}
	return d.Seconds() * float64(limit)
}

说明2

type Reservation struct {
	ok        bool         // 是否拿到了token
	lim       *Limiter     // 令牌桶地址
	tokens    int          // 令牌桶剩余的token数
	timeToAct time.Time    // 发放令牌的时间(可能是未来时间)
	limit Limit            // qps
}

func (r *Reservation) Cancel() {
	r.CancelAt(time.Now())
}

func (r *Reservation) CancelAt(now time.Time) {
	if !r.ok {
		return
	}
        // 上锁
	r.lim.mu.Lock()
	defer r.lim.mu.Unlock()
        // 这种情况 都不用处理直接返回
	if r.lim.limit == Inf || r.tokens == 0 || r.timeToAct.Before(now) {
		return
	}

	// calculate tokens to restore
	// The duration between lim.lastEvent and r.timeToAct tells us how many tokens were reserved
	// after r was obtained. These tokens should not be restored.
	restoreTokens := float64(r.tokens) - r.limit.tokensFromDuration(r.lim.lastEvent.Sub(r.timeToAct))
	if restoreTokens <= 0 {
		return
	}
	// advance time to now
	now, _, tokens := r.lim.advance(now)
	// calculate new number of tokens
	tokens += restoreTokens
	if burst := float64(r.lim.burst); tokens > burst {
		tokens = burst
	}
	// update state
	r.lim.last = now
	r.lim.tokens = tokens
	if r.timeToAct == r.lim.lastEvent {
		prevEvent := r.timeToAct.Add(r.limit.durationFromTokens(float64(-r.tokens)))
		if !prevEvent.Before(now) {
			r.lim.lastEvent = prevEvent
		}
	}
}
