func determineEncoding(r *bufio.Reader) encoding.Encoding {
 bytes ,err := r.Peek(1024)
 if err != nil {
  log.Printf("Fetcher error : %v",err)
  return unicode.UTF8
 }

 e, _, _ := charset.DetermineEncoding(bytes,"")
 return e
}
