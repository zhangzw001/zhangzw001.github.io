parsePattern 函数

// Only one * is allowed
// /test/v1/add -> [test v1 add]
// /test/:version/add -> [test :version add]
// /test/*name/add -> [test *name]
func parsePattern(pattern string ) []string {
	vs := strings.Split(pattern,"/")
	parts := make([]string,0)
	for _ , item := range vs {
		if item != "" {
			parts = append(parts, item)
			if item[0] == '*' {
				break
			}
		}
	}
	return parts
}

路由分析测试


func TestParsePattern(t *testing.T) {
	tests := []struct {
		patterns string
		parts []string
	}{
		{"/test/v1/add",[]string{"test", "v1", "add"}},
		{"/test/*name/add",[]string{"test", "*name"}},
		{"/test/:name",[]string{"test",":name"}},
		{"/test/:name/abc",[]string{"test",":name","abc"}},
	}

	m := make(map[string][]string)

	for _, test := range tests {
		m[test.patterns] = test.parts
		p := parsePattern(test.patterns)
		if !reflect.DeepEqual(p, m[test.patterns]){
			t.Errorf("expect %v, but actual %v\n",m[test.patterns],p)
		}
	}

}
