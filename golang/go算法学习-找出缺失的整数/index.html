Go算法学习 找出缺失的整数

原文: 程序员小灰 找出缺失的整数

实现代码

package main

import (
	"fmt"
	"math/rand"
	"time"
)

var (
	a []int
	b []int
	c []int
	l = 100
	r = rand.New(rand.NewSource(time.Now().UnixNano()))

)

func init() {
	// 1~10 随机数
	tmp := r.Intn(l-1)+1
	// 在生成一个不等于tmp的随机数
	var tmp2 int
	for tmp2 == 0 || tmp2 == tmp {
		tmp2 = r.Intn(l-1)+1
	}

	fmt.Println(tmp,tmp2)
	for i:=1 ;i <=l;i++ {
		if i == tmp {
			b = append(b, i)
			continue
		}
		a = append(a, i)
		b = append(b, i)
		b = append(b, i)

	}

	for i:=1 ;i <=l;i++ {
		if i == tmp ||i == tmp2  {
			c = append(c,i)
			continue
		}
		c = append(c,i)
		c = append(c,i)
	}
	//fmt.Println(">>> 有序数组")
	//fmt.Println(a)
	//fmt.Println(b)
	//fmt.Println(c)
	random(a)
	random(b)
	random(c)
	//fmt.Println(">>> 无序数组")
	//fmt.Println(a)
	//fmt.Println(b)
	//fmt.Println(c)
}

// 随机打乱字符串算法
//Fisher-Yates随机置乱算法
//也称高纳德置乱算法
func random(nums []int) {
	if len(nums) <= 1 {
		return
	}

	for len(nums)>0 {
		n := len(nums)
		tmp := r.Intn(n)
		nums[tmp] , nums[n-1] = nums[n-1], nums[tmp]
		//对前n-1进行替换
		nums = nums[:n-1]
	}
}

//题目：一个无序数组里有99个不重复正整数，范围从1到100，唯独缺少一个整数。如何找出这个缺失的整数？
func  findInt1(a []int ) int  {
	sum := 0
	for i:= 0 ;i < l-1 ;i++ {
		sum+=a[i]
	}
	return (1+l)*l/2-sum
}

//题目扩展：一个无序数组里有若干个正整数，范围从1到100，其中99个整数都出现了偶数次，
//只有一个整数出现了奇数次（比如1,1,2,2,3,3,4,5,5），如何找到这个出现奇数次的整数？
// 思路: 所有的数做 异或
func findInt2(a []int ) int {
	rst := a[0]
	for i:= 1 ; i < len(a);i++{
		rst ^=a[i]
	}
	return rst
}

//题目第二次扩展：一个无序数组里有若干个正整数，范围从1到100，其中98个整数都出现了偶数次，
//只有两个整数出现了奇数次（比如1,1,2,2,3,4,5,5），如何找到这个出现奇数次的整数？
// 思路:
// 1. 先对所有数异或, 异或的结果就是两个奇数的异或结果, 这个数二进制位必定有一位是1
// 2. 按异或结果其中一个1的位置, 将无序数组分成两个数组
//
func findInt3(a []int) (int,int) {
	a1 := findInt2(a)
	//fmt.Println(a1)
	m := 0
	for a1 & 1 == 0 {
		a1 >>= 1
		m++
	}
	//fmt.Println(m)
	odd,even := 0,0
	for i:=0; i < len(a); i++ {
		// 说明该位 是1
		if a[i] >> m & 1 == 1 {
			odd ^=a[i]
		}else {
			even ^=a[i]
		}
	}
	return odd,even
}
func main() {
	fmt.Println(findInt1(a))
	fmt.Println(findInt2(b))
	fmt.Println(findInt3(c))
}
