实现 Raft Leader选举和心跳检测

首先对Raft-Extended的相关几点做个说明

1. raft集群在 ticker 超时 后进行选举, 节点未收到心跳检测就会发起选举
2. 选举的节点会存在几种状态切换
    follower  -> candidate : 开始选举的时候
    candidate -> leader    : 选举成功的时候
    candidate -> follower  : 选举失败的时候
    leader    -> follower  : 旧的过期节点
    对于 candidate -> candidate的情况, 这里是被分解成 candidate -> follower -> candidate
3. raft节点会在延迟不同的时间后 在切换成 candidate, 防止多节点同时切换
4. raft节点在成为 candidate 后, 会并发向每个节点发一次 RequestVote 来进行投票, 然后等到有半数同意就成为leader 并启动goroutine 进行持续心跳检测(AppendEntries RPC), 如果没有人同意那么就没人成为leader会等到下次超时发起新的选举
    在成为leader之前, raft节点同样可能收到别的 旧leader发来的AppendEntries RPC, 所以在 AppendEntries RPC 方法里面,我们需要如下判断:
    i) 如果收到的任期号小于当前任期号, 那么直接拒绝忽略
    ii) 如果收到的任期号大于等于那就做如下修改:
       把自己的状态改为 follower, 记录候选人id, 领导人id , 修改自己的当前任期号 和已提交任期号

type Log struct {
	mu	 sync.Mutex
	leaderID   int	   // leaderID, 表示当前集群已经选举完成
	votedFor   int	   // 当前获得选票的候选人id, -1表示没有人获得选票, 用于一个节点值会给一个候选人投票
	role       int	   // 节点处在什么角色, follower,candidate,leader
	currTerm   int	   // 当前任期号
	eleRelay   time.Duration // 选举定时器
	ticker     *time.Ticker
	isElect    chan struct{}
}

var (
	r	   = newRand()
	newDalyFunc = func(i, vari int) time.Duration {
		return time.Millisecond * time.Duration(i+r.Intn(vari))
	}
)

选举主进程 Election

func (rf *Raft) Election() {
	for {
		select {
		// 收到选举的请求
		case <-rf.log.isElect:
			// 开始选举
			go rf.startElect()
		// 定时器超时, 没有收到心跳检测
		case <-rf.log.ticker.C:
			// 需要选举
			go func() {
				rf.log.isElect <- struct{}{}
			}()
		default:
		}
	}
}

选举方法

func (rf *Raft) startElect() {
	// 发送投票 并提交
	rf.log.mu.Lock()
	rf.log.ticker.Stop()
	// 并且设置所有 votedFor 都为-1
	rf.log.convRole(follower, -1, rf.log.currTerm)
	rf.log.leaderID = -1
	rf.log.eleRelay = newDalyFunc(50, 200)
	currTerm := rf.log.currTerm
	rf.log.mu.Unlock()
	// 发送投票 并提交
	if rf.log.role == follower {
		// 新的任期号
		nextTerm := currTerm + 1
		// 进行随机的延迟
		time.Sleep(rf.log.eleRelay)

		// 多节点可能同时想要成为 candidate, leaderID如果不等于-1, 可能已经完成了选举, 选举随机时间慢的节点就不用在去选举了
		if rf.log.leaderID == -1 {
			rf.log.mu.Lock()
			// 成为候选人
			rf.log.convRole(candidate, rf.me, nextTerm)
			rf.log.mu.Unlock()

			var agrees int32 = 1
			wg := &sync.WaitGroup{}

			for id, _ := range rf.peers {
				if rf.me == id {
					continue
				}
				// 并发发送
				wg.Add(1)
				go func(id int) {
					defer wg.Done()
					args := &RequestVoteArgs{Xterm: nextTerm, CandidateId: rf.me, LastLogTerm: currTerm}
					reply := &RequestVoteReply{}
					// rpc请求投票, 同时告知其他人 自己是候选人
					if ok := rf.sendRequestVote(id, args, reply); ok {
						// 如果节点当前任期号小于其他节点,那么自己不配成为候选人
						if reply.Term > rf.log.currTerm {
							rf.log.mu.Lock()
							rf.log.convRole(follower, -1, reply.Term)
							rf.log.mu.Unlock()
							return
						}
						// 如果是赞成票
						if reply.Ok {
							atomic.AddInt32(&agrees, 1)
							if agrees > int32(len(rf.peers)/2) && rf.log.role == candidate {
								rf.log.mu.Lock()
								rf.log.convRole(leader, 0, 0)
								rf.log.mu.Unlock()
								// 心跳
								go rf.HeartBeat()
							}
						}
					}
				}(id)
			}
			// 可能没有选出来leader
			go func() {
				wg.Wait()
				if agrees <= int32(len(rf.peers)/2) {
					rf.log.mu.Lock()
					rf.log.convRole(follower,-1,rf.log.currTerm)
					rf.log.mu.Unlock()
					rf.log.ticker.Reset(newDalyFunc(300, 1))
				}
			}()
		}
	}

}
  • 设置状态

func (l *Log) setRole(role int) {
	l.role = role
}
func (l *Log) setVotedFor(votedFor int) {
	l.votedFor = votedFor
}
func (l *Log) setCurrTerm(xterm int) {
	l.currTerm = xterm
}

func (l *Log) convRole(role, votedFor, xterm int) {
	switch role {
	// 1. candidate -> follower : 候选人没有获得半数的投票
	// 2. leader -> follower : 过期的leader,或者xterm高于currTerm
	case follower:
		l.setRole(follower)
		l.setVotedFor(votedFor)
		l.setCurrTerm(xterm)
	// 1. follower -> candidate : 集群没有leader,或者没收到心跳检测
	// 2. candidate -> candidate : 没有选出leader(比如集群数量少于半数)
	case candidate:
		// candidate 需要
		l.setRole(candidate)
		l.setVotedFor(votedFor)
		l.setCurrTerm(xterm)
	// 1. candidate -> leader : 得到过半选票
	case leader:
		// leader只需要修改第一个角色即可
		l.setRole(leader)
	default:
		log.Println(" 不支持的角色")
	}
}
  • 投票的rpc方法 RequestVote
func (rf *Raft) sendRequestVote(server int, args *RequestVoteArgs, reply *RequestVoteReply) bool {
	ok := rf.peers[server].Call("Raft.RequestVote", args, reply)
	return ok
}

func (rf *Raft) RequestVote(args *RequestVoteArgs, reply *RequestVoteReply) {
	// Your code here (2A, 2B).
	rf.log.mu.Lock()
	defer rf.log.mu.Unlock()
	// 当前rf.me是有可能成为 candidate, 并且同样可能收到投票rpc
	reply.Term = rf.log.currTerm

	// 你当前任期号比我的还小, 你发什么rpc?
	if args.LastLogTerm < rf.log.currTerm {
		reply.Ok = false
		return
	}
	// 以前是leader, 下线了, 重新连上来的时候, 集群中最新的 LastLogTerm 已经比自己的大了
	if args.LastLogTerm > rf.log.currTerm {
		rf.log.convRole(follower, -1, args.LastLogTerm)
	}
	// 等于的情况本身是要判断追加日志长度是否 大于等于,所以这里先直接改为 follower, 否则多个 candidate 拥有相同的 LastLogTerm 就都保持 candidate 角色了, 我们要求只让一个成为 candidate
	// 这里暂时还没有 appendEntries 长度, 之后在添加
	if args.LastLogTerm == rf.log.currTerm {
		rf.log.convRole(follower, -1, args.LastLogTerm)
	}
	// 只有没有 候选人id的才有资格投票
	if rf.log.votedFor == -1 {
		reply.Ok = true
		rf.log.votedFor = args.CandidateId
	}

	return
}

leader节点的心跳检测 HeartBeat

func (rf *Raft) HeartBeat() {
	for {
		// 期间有可能 从leader 变成 follower
		if rf.log.role != leader {
			return
		}
		for id, _ := range rf.peers {
			if id == rf.me {
				// leader 重置心跳超时定时器, follower 是由心跳检测的rpc重置
				rf.log.ticker.Reset(newDalyFunc(200, 100))
				continue
			}
			// 成功了就会重置超时时钟
			// 失败了就不管, 等超时时钟触发即可开始选举
			go rf.heartBeat(id)
		}
		time.Sleep(newDalyFunc(100, 1))
	}

}

func (rf *Raft) heartBeat(id int) {
	if id == rf.me {
		return
	}
	arg := RequestVoteArgs{Xterm: rf.log.currTerm, CandidateId: rf.me}
	reply := RequestVoteReply{}
	if ok := rf.sendAppendEntries(id, &arg, &reply); ok {
		// 如果 节点返回的任期号更大, 说明我是个 过期的旧leader -> follower
		// 如果我是个旧的leader,那集群目前很有可能已经有leader了, 那么我的 votedFor和leaderID也是需要更新, 那我应该在哪更新呢?
		if reply.Term > rf.log.currTerm {
			rf.log.mu.Lock()
			rf.log.convRole(follower, -1, reply.Term)
			rf.log.leaderID = -1
			rf.log.mu.Unlock()
		}
	}
}
  • 心跳检测的rpc AppendEntries
func (rf *Raft) sendAppendEntries(server int, args *RequestVoteArgs, reply *RequestVoteReply) bool {
	ok := rf.peers[server].Call("Raft.AppendEntries", args, reply)
	return ok
}

func (rf *Raft) AppendEntries(args *RequestVoteArgs, reply *RequestVoteReply) {
	// Your code here (2A, 2B).
	rf.log.mu.Lock()
	defer rf.log.mu.Unlock()
	reply.Term = rf.log.currTerm
	// 这里 rf.me 有可能刚好是 candidate 状态, 并且刚好收到了另外的 自称为leader的节点发送的 rpc
	// 这个心跳或者追加日志的任期号肯定不能小于当前任期号,
	// 等于的情况是持续的心跳检测, 大于的情况是刚成为leader第一次的append日志
	if args.Xterm < rf.log.currTerm {
		reply.Ok = false
		return
	}
	// 以下就是大于等于的情况
	rf.log.ticker.Reset(newDalyFunc(300, 1))
	// 这里就要设置为 follower , 因为自己有可能在 candidate 状态
	rf.log.convRole(follower, args.CandidateId, args.Xterm)
	rf.log.leaderID = args.CandidateId
	reply.Ok = true
	return
}
  • 遇到的几个问题
1. 在哪个时间点将 follower 切换成 candidate ?
答: 首先让 follower 延迟随机时间 , 然后需要先判断当前集群是否 已经选举出了leader, 在没有leader的的情况下让当前节点成为 candidate.  当延迟更久的 follower 发现当前集群已经选举出了leader, 所以直接放弃选举了
2. 当节点数量越多时,就会出现更多的分票的情况,导致选举不成功
答: 这个问题是我 投票条件漏写了 args.LastLogTerm == rf.log.currTerm 等于的情况, 导致出现了多个 candidate, 从而出现了各个 candidate都给自己投票
3. 为什么用ticker来超时, 还不适用time.Now来按时间分割任期号的有效期呢
答: time.Now依赖各个节点时钟是否一致, 可能导致一些不可预料的错误
