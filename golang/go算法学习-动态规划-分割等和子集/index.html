//416. 分割等和子集
//题目链接：https://leetcode-cn.com/problems/partition-equal-subset-sum/
//
//题目难易：中等
//
//给定一个只包含正整数的非空数组。是否可以将这个数组分割成两个子集，使得两个子集的元素和相等。
//
//注意: 每个数组中的元素不会超过 100 数组的大小不会超过 200
//
//示例 1: 输入: [1, 5, 11, 5] 输出: true 解释: 数组可以分割成 [1, 5, 5] 和 [11].   示例 2: 输入: [1, 2, 3, 5] 输出: false 解释: 数组不能分割成两个元素和相等的子集.

// 1. 暴力回溯法
// 1,5,11,5
// 找到所有和=22, 取一半 = 11
// 先取第一个数字: 1,1+5,1+11,1+5|1+5+11,1+5+5|1+11+5  存在==11
// 先取第二个数字: 5+11,5+5
// 先取第三个数字: 11+5
// 先取第四个数字: 5
// 回溯遍历可以得到结果, 但是会超时
// 时间复杂度: O(n*n!) ?

func canPartition(nums []int) bool {
	sum := 0
	for i := 0; i < len(nums); i++ {
		sum += nums[i]
	}
	if sum%2 != 0 {
		return false
	}

	// 回溯三部曲
	// 1. 传参:
	// 2. 终止条件
	// 3. 单次循环
	part := 0
	result := false
	var dfs func(index int)
	dfs = func(index int) {
		// 终止条件
		if part > sum/2 {
			return
		}
		if part == sum/2 {
			result = true
			return
		}
		// 单次循环,每次取值进行累加
		for i := index; i < len(nums)-1; i++ {
			part += nums[i]
			// 递归
			dfs(i + 1)
			// 回溯
			part -= nums[i]
		}
	}
	dfs(0)
	return result
}

// 01背包问题, 元素只能使用一次
// 本次是要求找出 元素和 等于总和/2
// 1. dp数组的定义
// dp[value] 表示 容量为value的背包,所背的物品价值为dp[value]
// 在这里
// dp[value] 表示 容量为value的背包,所存储的最大的值 dp[value]
// 2. dp数组的推导公式
// dp[value] = max(dp[value], dp[value-nums[j]]+nums[j]
// 3. dp数组初始化
// dp[0] = 0
// 4. dp数组的顺序
// 在动态规划：关于01背包问题，你该了解这些！（滚动数组）中就已经说明：如果使用一维dp数组，物品遍历的for循环放在外层，遍历背包的for循环放在内层，且内层for循环倒叙遍历！
// 5. 举例数组
// dp[value]的数值一定小于等于value, 肯定不能超过最大容量
// 如果dp[value]== value,说明子集合刚好可以凑成总和i, 这里我们的i上限 其实等于 sum/2
// 例如 输入[1,5,11,5], sum/2 = 11
// dp[value] = 0,1,1,1,1,5,6,6,6,6,10,11
func canPartition2(nums []int) bool {
	sum := 0
	for i := 0; i < len(nums); i++ {
		sum += nums[i]
	}
	if sum & 1 == 1  {
		return false
	}
	// max方法
	var max func(int,int) int
	max = func(a,b int) int {
		if a > b {
		return a
	}
		return b
	}

	target := sum >> 1
	// 1. 定义dp数组
	dp := make([]int, target+1)
	// 3. 初始化
	dp[0] = 0
	// 4. dp数组的顺序
	for i := 0; i < len(nums); i++ {
		// 每一个元素一定是不可重复放入，所以从大到小遍历
		for value := target; value >= nums[i]; value-- {
			dp[value] = max(dp[value], dp[value-nums[i]]+nums[i])
		}
	}
	// 等于target 说明整合凑成
	if dp[target] == target {
		return true
	}
	return false
}



// 优化版本, 不需要计算值
func canPartition3(nums []int) bool {
	sum := 0
	for i := 0; i < len(nums); i++ {
		sum += nums[i]
	}
	if sum & 1 == 1  {
		return false
	}
	target := sum >> 1
	// 1. 定义dp数组
	dp := make([]bool, target+1)
	// 3. 初始化
	dp[0] = true
	// 4. dp数组的顺序
	for i := 0; i < len(nums); i++ {
		// 每一个元素一定是不可重复放入，所以从大到小遍历
		for value := target; value >= nums[i]; value-- {
			dp[value] = dp[value]|| dp[value-nums[i]]
		}
		// 一旦等于target
		if dp[target] {
			return true
		}
	}
	return false
}

}
